142 # @logger()
143 def p_factor_sqrt(self, p):
144 """factor : factor SQRT factor"""
145 if p[3] < 0: # number is negative
146 if p[1] % 2 == 0: # base is even -> error
147 p[0] = None
148 raise ValueError
149 if p[1] % 2 != 0: # base is not even
150 if p[1].is_integer(): # number is integer, save '-' operator
151 p[0] = - round(abs(p[3] ** (1 / p[1])), 10)
152 else:
153 p[0] = round(abs(p[3] ** (1 / p[1])), 10)
154 else:
155 p[0] = p[3] ** (1 / p[1])
